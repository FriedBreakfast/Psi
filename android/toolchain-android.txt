include(CMakeForceCompiler)

set(ANDROID_STANDALONE "" CACHE FILEPATH "Path to Android standalone toolchain")
set(ANDROID_PROCESSOR "arm" CACHE STRING "Name to use for CMAKE_SYSTEM_PROCESSOR")

set(ANDROID TRUE)
set(CMAKE_CROSSCOMPILING TRUE)
set(CMAKE_SYSTEM_NAME Linux)
set(CMAKE_SYSTEM_VERSION 1)
set(CMAKE_SYSTEM_PROCESSOR ${ANDROID_PROCESSOR})

set(CMAKE_FIND_ROOT_PATH "${ANDROID_STANDALONE}")
set(CMAKE_FIND_ROOT_PATH_MODE_PROGRAM BOTH)
set(CMAKE_FIND_ROOT_PATH_MODE_INCLUDE ONLY)
set(CMAKE_FIND_ROOT_PATH_MODE_LIBRARY ONLY)

find_program(ANDROID_C_COMPILER NAMES arm-linux-androideabi-gcc ONLY_CMAKE_FIND_ROOT_PATH)
find_program(ANDROID_CXX_COMPILER NAMES arm-linux-androideabi-g++ ONLY_CMAKE_FIND_ROOT_PATH)

cmake_force_c_compiler(${ANDROID_C_COMPILER} GNU)
cmake_force_cxx_compiler(${ANDROID_CXX_COMPILER} GNU)
